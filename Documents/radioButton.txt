// Main_activity.java
import android.os.Bundle;
import android.app.Activity;
import android.view.View;
import android.widget.Button;
import android.widget.RadioButton;
import android.widget.Toast;

public class MainActivity extends Activity {

    private RadioButton select1;
    private RadioButton select2;
    private RadioButton select3;
    private Button resultButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Initializing RadioButtons
        select1 = findViewById(R.id.radioButton1);
        select2 = findViewById(R.id.radioButton2);
        select3 = findViewById(R.id.radioButton3);

        // Initializing Button
        resultButton = findViewById(R.id.button1);

        // Set OnClickListener on the Button
        resultButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                showResult();
            }
        });
    }

    private void showResult() {
        String message = null;

        // Checking which RadioButton is selected
        if (select1.isChecked()) {
            message = select1.getText().toString();
        } else if (select2.isChecked()) {
            message = select2.getText().toString();
        } else if (select3.isChecked()) {
            message = select3.getText().toString();
        }

        // Setting message if no option is selected
        message = (message == null) ? "You selected nothing" : "You have selected: " + message;

        // Displaying a Toast message
        Toast.makeText(this, message, Toast.LENGTH_LONG).show();
    }
}
